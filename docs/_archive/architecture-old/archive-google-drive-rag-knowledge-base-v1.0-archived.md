## 1. Executive Summary

### 1.1 Background

GenesisPod е·Іе®һзҺ° Google Drive еҹәзЎҖйӣҶжҲҗпјҲиҝһжҺҘгҖҒеҜје…ҘгҖҒеҜјеҮәпјүпјҢз”ЁжҲ·еҸҜд»Ҙе°Ҷдә‘з«Ҝж–ҮжЎЈеҜје…ҘеҲ°иө„жәҗеә“гҖӮз„¶иҖҢпјҢеҪ“еүҚеҜје…Ҙзҡ„еҶ…е®№д»…дҪңдёәйқҷжҖҒиө„жәҗеӯҳеӮЁпјҢж— жі•иў« AI зі»з»ҹжҷәиғҪжЈҖзҙўе’ҢеҲ©з”ЁгҖӮ

з”ЁжҲ·зҡ„ж ёеҝғиҜүжұӮжҳҜпјҡ**е°Ҷ Google Drive дёӯзҡ„дё“дёҡж–ҮжЎЈпјҲеҰӮиЎҢдёҡжҠҘе‘ҠгҖҒеҶ…йғЁзҹҘиҜҶеә“гҖҒз ”з©¶иө„ж–ҷпјүдҪңдёә AI зҡ„йўҶеҹҹзҹҘиҜҶиҫ“е…ҘпјҢдҪҝеҗҺз»ӯзҡ„ AI AskпјҲжҷәиғҪй—®зӯ”пјүе’Ң AI StudioпјҲж·ұеәҰз ”з©¶пјүиғҪеӨҹеҹәдәҺиҝҷдәӣз§ҒжңүзҹҘиҜҶз”ҹжҲҗжӣҙеҮҶзЎ®гҖҒжӣҙдё“дёҡзҡ„еӣһзӯ”е’ҢжҠҘе‘ҠгҖӮ**

иҝҷжҳҜ RAGпјҲRetrieval-Augmented Generationпјүжһ¶жһ„зҡ„е…ёеһӢеә”з”ЁеңәжҷҜпјҡ

- **Retrieval**пјҡд»Һз”ЁжҲ·зҡ„ Google Drive ж–ҮжЎЈдёӯжЈҖзҙўзӣёе…ідҝЎжҒҜ
- **Augmented**пјҡе°ҶжЈҖзҙўеҲ°зҡ„еҶ…е®№дҪңдёәдёҠдёӢж–Үеўһејә AI зҡ„еӣһзӯ”
- **Generation**пјҡеҹәдәҺеўһејәеҗҺзҡ„дёҠдёӢж–Үз”ҹжҲҗй«ҳиҙЁйҮҸиҫ“еҮә

### 1.4 Non-Goals

- **е®һж—¶еҚҸдҪңзј–иҫ‘**пјҡдёҚеңЁ DeepDive еҶ…зӣҙжҺҘзј–иҫ‘ Google Drive ж–ҮжЎЈ

- **и·Ёз”ЁжҲ·зҹҘиҜҶе…ұдә«**пјҡжң¬жңҹдёҚж”ҜжҢҒе°ҶдёӘдәәзҹҘиҜҶеә“е…ұдә«з»ҷе…¶д»–з”ЁжҲ·
- **йқһж–Үжң¬еҶ…е®№**пјҡдёҚж”ҜжҢҒеӣҫзүҮгҖҒи§Ҷйў‘гҖҒйҹійў‘зҡ„иҜӯд№үзҗҶи§ЈпјҲд»…ж”ҜжҢҒж–Үжң¬жҸҗеҸ–пјү
- **еӨҡиҜӯиЁҖж··еҗҲ**пјҡдёҚж”ҜжҢҒеҚ•дёӘж–ҮжЎЈдёӯеӨҡз§ҚиҜӯиЁҖзҡ„еҲҶеқ—дјҳеҢ–

model KnowledgeBase {

model KnowledgeBaseDocument {

  // жқҘжәҗдҝЎжҒҜ

  // ж–ҮжЎЈдҝЎжҒҜ

  mimeType        String   @map("mime_type")

  // зҙўеј•зҠ¶жҖҒ
  indexStatus     DocumentIndexStatus @default(PENDING)
  indexError      String?  @map("index_error")
  chunkCount      Int      @default(0) @map("chunk_count")
  tokenCount      Int      @default(0) @map("token_count")

  // еҗҢжӯҘдҝЎжҒҜ
  sourceModifiedAt DateTime? @map("source_modified_at")
  lastIndexedAt   DateTime? @map("last_indexed_at")
  contentHash     String?  @map("content_hash")  // з”ЁдәҺжЈҖжөӢеҶ…е®№еҸҳеҢ–

  // ж—¶й—ҙжҲі
  createdAt       DateTime @default(now()) @map("created_at")
  updatedAt       DateTime @updatedAt @map("updated_at")

  // е…ізі»

  @@unique([knowledgeBaseId, sourceId])
  @@index([knowledgeBaseId])
  @@map("knowledge_base_documents")
}

  RESOURCE

enum DocumentIndexStatus {

  PROCESSING
  INDEXED
  FAILED

  // дҪҚзҪ®дҝЎжҒҜ
  position     Int      // еқ—еңЁж–ҮжЎЈдёӯзҡ„йЎәеәҸ
  pageNumber   Int?     @map("page_number")
  sectionTitle String?  @map("section_title")

  // е…ғж•°жҚ®
  metadata     Json?

  // ж—¶й—ҙжҲі
  createdAt    DateTime @default(now()) @map("created_at")

  // е…ізі»
  document     KnowledgeBaseDocument @relation(fields: [documentId], references: [id], onDelete: Cascade)
  embedding    Embedding?

  @@index([documentId])
  @@map("chunks")
}

  createdAt DateTime @default(now()) @map("created_at")

  // е…ізі»
  chunk     Chunk    @relation(fields: [chunkId], references: [id], onDelete: Cascade)

  @@map("embeddings")
