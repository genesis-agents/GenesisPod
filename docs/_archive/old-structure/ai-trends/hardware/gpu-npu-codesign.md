### 2. 软件栈设计

```python
# 异构调度示例
class HeterogeneousScheduler:
    """
    GPU-NPU 异构调度器
    根据任务特点选择合适的硬件执行
    """

    def __init__(self, gpu, npu):
        self.gpu = gpu
        self.npu = npu
        self.profiler = HardwareProfiler()

    def schedule(self, model, input_data):
        # 分析模型结构
        ops = self.analyze_model(model)

        execution_plan = []
        for op in ops:
            # 根据算子类型和输入规模选择硬件
            if self.should_run_on_npu(op):
                execution_plan.append(("npu", op))
            else:
                execution_plan.append(("gpu", op))

        return execution_plan

    def should_run_on_npu(self, op):
        """决策逻辑"""
        # NPU 支持且效率更高的算子
        npu_efficient_ops = [
            "conv2d", "matmul", "softmax",
            "layer_norm", "attention"
        ]

        # 检查算子类型
        if op.type not in npu_efficient_ops:
            return False

        # 检查输入规模 (小 batch 可能 GPU 更优)
        if op.batch_size < self.npu.min_efficient_batch:
            return False

        # 检查精度要求
        if op.requires_fp32:
            return False

        return True
```

## 软硬件协同设计

### 1. 编译器优化

```python
# AI 编译器工作流程
class AICompiler:
    """
    AI 编译器负责将高级模型转换为硬件高效代码
    """

    def compile(self, model, target_hardware):
        # 1. 前端: 将 PyTorch/TF 模型转为 IR
        ir = self.frontend.parse(model)

        # 2. 图优化
        ir = self.optimize_graph(ir)

        # 3. 算子调度
        schedule = self.generate_schedule(ir, target_hardware)

        # 4. 代码生成
        code = self.backend.codegen(schedule, target_hardware)

        return code

    def optimize_graph(self, ir):
        """图级优化"""
        # 常量折叠
        ir = constant_folding(ir)
        # 算子融合
        ir = operator_fusion(ir)
        # 死代码消除
        ir = dead_code_elimination(ir)
```
